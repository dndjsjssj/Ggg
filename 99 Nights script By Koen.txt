local missingChildren = {

local espEnabled = false
local espFolder = Instance.new("Folder")
espFolder.Name = "ChestESP"
espFolder.Parent = player:WaitForChild("PlayerGui")
 
local highlightParts = {}
 
MainTab:CreateToggle({
    Name = "Toggle Chest ESP",
    CurrentValue = false,
    Flag = "ChestESP",
    Callback = function(state)
        espEnabled = state
 
        if espEnabled then
            local chestsFolder = workspace:FindFirstChild("Chests")
            if not chestsFolder then
                warn("No Chests folder found in workspace.")
                return
            end
 
            for _, chest in pairs(chestsFolder:GetChildren()) do
                if chest:IsA("Model") and chest.PrimaryPart then
                    local highlight = Instance.new("BoxHandleAdornment")
                    highlight.Adornee = chest.PrimaryPart
                    highlight.Size = chest.PrimaryPart.Size + Vector3.new(0.5, 0.5, 0.5)
                    highlight.Color3 = Color3.new(1, 0.5, 0)
                    highlight.Transparency = 0.4
                    highlight.AlwaysOnTop = true
                    highlight.Parent = espFolder
                    table.insert(highlightParts, highlight)
                end
            end
        else
            for _, h in pairs(highlightParts) do
                h:Destroy()
            end
            highlightParts = {}
        end
    end
})
 
MainTab:CreateButton({
    Name = "Spawn All Resources",
    Callback = function()
        local resourcesFolder = workspace:FindFirstChild("Resources")
        if not resourcesFolder then
            warn("Resources folder not found in workspace.")
            return
        end
 
        local playerRoot = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
        if not playerRoot then
            warn("Player character not loaded.")
            return
        end
 
        local spawnPos = playerRoot.Position + Vector3.new(0, 0, 5)
 
        local resourceNames = {"Logs", "Coals", "Saplings", "BrokenMicrowave"}
 
        for _, resName in pairs(resourceNames) do
            local resModel = resourcesFolder:FindFirstChild(resName)
            if resModel then
                local clone = resModel:Clone()
                clone.Parent = workspace
                if clone.PrimaryPart then
                    clone:SetPrimaryPartCFrame(CFrame.new(spawnPos))
                else
                    clone:PivotTo(CFrame.new(spawnPos))
                end
                spawnPos = spawnPos + Vector3.new(3, 0, 0)
            else
                warn("Resource "..resName.." not found in Resources folder.")
            end
        end
    end
})
